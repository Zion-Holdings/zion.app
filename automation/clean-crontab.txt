# Clean Automation Cron Configuration
# This configuration only runs working automation systems

# Set environment variables
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin:/usr/sbin:/sbin:/opt/homebrew/bin
HOME=/Users/miami2
NVM_DIR=/Users/miami2/.nvm

# Load NVM and Node.js in cron environment
@reboot source /Users/miami2/.nvm/nvm.sh && cd /Users/miami2/Documents/GitHub/bolt.new.zion.app/automation && node clean-automation-controller.js start

# Start working automation systems every 6 hours
0 */6 * * * source /Users/miami2/.nvm/nvm.sh && cd /Users/miami2/Documents/GitHub/bolt.new.zion.app/automation && node clean-automation-controller.js start

# Health check every hour
0 * * * * source /Users/miami2/.nvm/nvm.sh && cd /Users/miami2/Documents/GitHub/bolt.new.zion.app/automation && node clean-automation-controller.js status

# Daily cleanup at 2 AM

# Weekly restart on Sunday at 3 AM
0 3 * * 0 source /Users/miami2/.nvm/nvm.sh && cd /Users/miami2/Documents/GitHub/bolt.new.zion.app/automation && node clean-automation-controller.js stop && sleep 30 && node clean-automation-controller.js start
